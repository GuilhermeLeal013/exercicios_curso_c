#include <stdlib.h>
#include <stdio.h>


int main()
{
    int cont;
    
    cont = 0;          /* inicialização da variavel contadora */
    while(cont <= 10)  /* teste do laço de repetição */

     {
          printf("%d \n", cont);
    
           cont++;    /* incremento da variavel contadora */
     }


    return 0;
}
